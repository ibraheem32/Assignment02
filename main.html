Adding lines & Annotations for Mean and Median¶

using plt.axvline() to add lines for mean and median.
using plt.text() to add annotations for mean and median.

In [103]:
# creating histogram again for mean & median
plt.figure(figsize=(8, 6))  # Set figure size
plt.hist(data, bins=30, edgecolor='black', color='skyblue')  # Create histogram with 30 bins

# adding lines for mean & median
plt.axvline(mean_value, color='red', linestyle='dashed', linewidth=2, label=f'Mean: {mean_value:.2f}')
plt.axvline(median_value, color='green', linestyle='dashed', linewidth=2, label=f'Median: {median_value:.2f}')

# annotations for mean & median
plt.text(mean_value, 20, f'Mean: {mean_value:.2f}', color='red', fontsize=12, horizontalalignment='right')
plt.text(median_value, 20, f'Median: {median_value:.2f}', color='green', fontsize=12, horizontalalignment='left')

# title
plt.title("Histogram with Annotations for Mean & Median", fontsize=16)

#labels
plt.xlabel("Value", fontsize=14)
plt.ylabel("Frequency", fontsize=14)

# legend
plt.legend()

# font size
plt.xticks(fontsize=14)
plt.yticks(fontsize=14)

# showing final plot
plt.show()
